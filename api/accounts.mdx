---
title: Get accounts
description: Retrieve a list of accounts with optional filtering, sorting, and pagination
---

## Endpoint

```
GET /api/v1/accounts
```

Retrieve all accounts for the authenticated user with support for filtering, sorting, and optional balance calculation.

## Query parameters

<ParamField query="type" type="string">
  Filter accounts by type (e.g., `checking`, `savings`, `credit`)
</ParamField>

<ParamField query="q" type="string">
  Search accounts by name using case-insensitive pattern matching
</ParamField>

<ParamField query="is_active" type="boolean" default="true">
  Filter by active status. Accepts `true`, `false`, `1`, or `0`
</ParamField>

<ParamField query="is_default" type="boolean" default="false">
  Filter by default account status. Accepts `true`, `false`, `1`, or `0`
</ParamField>

<ParamField query="include_balance" type="boolean" default="false">
  Include calculated balance for each account. Requires `as_of_date` or defaults to today
</ParamField>

<ParamField query="as_of_date" type="string">
  Date for balance calculation in YYYY-MM-DD format. Only used when `include_balance=true`
</ParamField>

<ParamField query="ids" type="string">
  Comma-separated list of account UUIDs to filter by
</ParamField>

<ParamField query="fields" type="string">
  Comma-separated list of fields to return. Available fields: `id`, `name`, `type`, `is_default`, `is_active`, `created_at`
</ParamField>

<ParamField query="sort" type="string" default="name">
  Field to sort by. Options: `name`, `created_at`, `type`
</ParamField>

<ParamField query="order" type="string" default="asc">
  Sort order. Options: `asc`, `desc`
</ParamField>

<ParamField query="limit" type="integer" default="100">
  Maximum number of results to return (1-1000)
</ParamField>

<ParamField query="offset" type="integer" default="0">
  Number of results to skip for pagination
</ParamField>

<ParamField query="cursor" type="string">
  Cursor for pagination. Cannot be used with `offset`
</ParamField>

## Response fields

<ResponseField name="data" type="array">
  Array of account objects

  <ResponseField name="id" type="string">
    Unique account identifier (UUID)
  </ResponseField>

  <ResponseField name="name" type="string">
    Account name
  </ResponseField>

  <ResponseField name="type" type="string">
    Account type (e.g., checking, savings, credit)
  </ResponseField>

  <ResponseField name="is_default" type="boolean">
    Whether this is the default account
  </ResponseField>

  <ResponseField name="is_active" type="boolean">
    Whether the account is active
  </ResponseField>

  <ResponseField name="created_at" type="string">
    Account creation timestamp (ISO 8601)
  </ResponseField>

  <ResponseField name="balance" type="number">
    Calculated account balance (only included when `include_balance=true`)
  </ResponseField>
</ResponseField>

<ResponseField name="meta" type="object">
  Pagination and metadata

  <ResponseField name="total" type="integer">
    Total number of matching accounts
  </ResponseField>

  <ResponseField name="returned" type="integer">
    Number of accounts returned in this response
  </ResponseField>

  <ResponseField name="limit" type="integer">
    Maximum results per page
  </ResponseField>

  <ResponseField name="offset" type="integer">
    Current offset
  </ResponseField>

  <ResponseField name="next_cursor" type="string | null">
    Cursor for the next page, or null if no more results
  </ResponseField>

  <ResponseField name="sort" type="string">
    Field used for sorting
  </ResponseField>

  <ResponseField name="order" type="string">
    Sort order applied
  </ResponseField>

  <ResponseField name="include_balance" type="boolean">
    Whether balances were included
  </ResponseField>

  <ResponseField name="as_of_date" type="string | null">
    Date used for balance calculation
  </ResponseField>
</ResponseField>

## Example request

```bash
curl -X GET "https://api.cashcat.app/api/v1/accounts?is_active=true&include_balance=true&as_of_date=2024-03-15" \
  -H "Authorization: Bearer YOUR_API_KEY"
```
